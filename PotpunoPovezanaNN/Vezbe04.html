def make_model(ulaz, izlaz):
    model = Sequential()
    model.add(Dense(200, input_dim=np.shape(ulaz)[1], activation='relu')) 
    model.add(Dense(100, activation='relu')) 
    model.add(Dense(50, activation='relu'))
    model.add(Dense(izlaz.shape[1], activation='softmax'))

    model.compile(optimizer='adam', 
                  loss='binary_crossentropy', 
                  metrics=['accuracy']) 
    
    return model

def model_perf(ulaz, izlaz, model):
    izlazPred = model.predict(ulaz)
    izlazPred = np.argmax(izlazPred, axis=1)
    izlaz = np.argmax(izlaz, axis=1)
    A = accuracy_score(izlaz, izlazPred)

    return A

In [12]:
def make_model(ulaz, izlaz):
    model = Sequential()
    model.add(Dense(200, input_dim=np.shape(ulaz)[1], activation='relu')) 
    model.add(Dense(100, activation='relu', kernel_regularizer=l2(0.01))) 
    model.add(Dense(50, activation='relu', kernel_regularizer=l2(0.01)))
    model.add(Dense(izlaz.shape[1], activation='softmax'))

    model.compile(optimizer='adam', 
                  loss='binary_crossentropy', 
                  metrics=['accuracy']) 
    
    return model
